Main menu class MainMenu

  package modules
  {
   import com.moto.template.common.events.ItemRendererEvent;
   import com.moto.template.common.events.MotoEvent;
   import com.moto.template.common.events.UpdateMotoObjectEvent;
   import com.moto.template.common.Moto;
   import com.moto.template.modules.menu.AbstractMenuItemRenderer;
   import com.moto.template.modules.menu.DynamicMotoMenu;
   import com.moto.template.shell.model.vo.MotoObjectVO;
     import flash.filters.DropShadowFilter;
   import flash.geom.Point;
   import flash.geom.Rectangle;
   import flash.display.MovieClip;
   import flash.display.Sprite;
   import flash.events.Event;
     public class MainMenu extends DynamicMotoMenu
   {
    private var _menu_holder_mc:MovieClip;
    private var _submenu_holder_mc:MovieClip;
    private var _preview_mc:MovieClip;
      public function MainMenu()
    {
     // Get elements
     _menu_holder_mc = this["menu_holder_mc"];
     _submenu_holder_mc = this["submenu_holder_mc"];
     _preview_mc = this["preview_mc"];
       _init();
       // Create listeners for complete and error module events
     addEventListener(INITIALIZATION_COMPLETE, _initializationHandler);
     addEventListener(INITIALIZATION_ERROR, _initializationHandler);
    }
      private function _init():void
    {
     // Setup menu items arranger
     // Set item renderers holder
     itemsArranger.autoUpdate = false;
     itemsArranger.target = _menu_holder_mc;
     itemsArranger.rows = 1;
     itemsArranger.horizontalSpacing = 4;
       subMenuRemoveDelay = 1;
       //Submenu hiding delay
     subMenuAutoHideTimeout = 1;
    }
      override public function init(vo:MotoObjectVO):void
    {
     super.init(vo);
     dispatchEvent(new UpdateMotoObjectEvent(UpdateMotoObjectEvent.MOTO_OBJECT_SIZE_UPDATED));
       // Create listener for the switch page logic to switch menu buttons
     // We should check if Moto is created cause there is no Moto in the
     // control panel
     if (!limitedMode)
      Moto.getInstance().eventDispatcher.addEventListener(MotoEvent.SWITCH_PAGE, _motoSwitchPageHandler);
    }
      private function _initializationHandler(event:Event):void
    {
     switch (event.type)
     {
      case INITIALIZATION_COMPLETE:

      break;
        case INITIALIZATION_ERROR:
       //Remove item arranger items if there are any
       itemsArranger.removeItems();
         // Show preview in the control panel
       _preview_mc.visible = limitedMode;
      break;
     }
    }
      override protected function initListControls(dataProvider:Array):void
   {
    super.initListControls(dataProvider);
       // Switch menu button
     _switchSelectedMenuButton();
       // Hide preview
     _preview_mc.visible = !(dataProvider.length || !limitedMode);
     _initializationSuccess();
   }
      override public function dispose():void
    {
     super.dispose();
     if (!limitedMode)
      Moto.getInstance().eventDispatcher.removeEventListener(MotoEvent.SWITCH_PAGE, _motoSwitchPageHandler)
    }
      private function _initializationSuccess():void
    {
     var maxWidth:Number = 0;
     var maxHeight:Number = 0;
     var item_btn:AbstractMenuItemRenderer;
     var itemDimensions:Rectangle;
       var i:int = itemsArranger.itemsCollection.length;
     while (i--)
     {
      item_btn = itemsArranger.itemsCollection[i];
      itemDimensions = item_btn.getDimensions();
      maxWidth = Math.max(maxWidth, itemDimensions.width);
      maxHeight = Math.max(maxHeight, itemDimensions.height);
     }
       var nextX:Number = 0;
       var length:int = itemsArranger.itemsCollection.length;
     for (i = 0; i < length; i++)
     {
      item_btn = itemsArranger.itemsCollection[i];
      item_btn.x = Math.round(nextX);
      item_btn.setSize(maxWidth, maxHeight);
      nextX += item_btn.getDimensions().width + itemsArranger.horizontalSpacing;
     }
    }
      override public function getDimensions():Rectangle
    {
     if (!itemsArranger.itemsCollection)
      return super.getDimensions();
     var w:Number = 0;
     var h:Number = 0;
     var i:int = itemsArranger.itemsCollection.length;
     var item_btn:AbstractMenuItemRenderer;
     while (i--)
     {
      item_btn = itemsArranger.itemsCollection[i];
      var itemDimensions:Rectangle = item_btn.getDimensions();
      w += itemDimensions.width + itemsArranger.horizontalSpacing;
      h = Math.max(h, itemDimensions.height);
     }
     w -= itemsArranger.horizontalSpacing;
     return new Rectangle(0, 0, w, h);
    }
      private function _motoSwitchPageHandler(event:MotoEvent):void {
     // Change selected menu item if website page was switched.
     _switchSelectedMenuButton();
    }
      private function _switchSelectedMenuButton():void {
     // Get menu item index if there is menu item that refers to the current page.
     if (!limitedMode)
      updateSelectedMenuButton(Moto.getInstance().currentPage);
    }
      override public function getSubMenuTargetHolder():Sprite {
     return _submenu_holder_mc;
    }
      override protected function itemClickHandler(event:ItemRendererEvent):void {
     // Do not allow clicks while animation is playing
     if (!Moto.getInstance().animationIsPlaying)
      super.itemClickHandler(event);
    }
      override public function showSubMenu(itemRenderer:AbstractMenuItemRenderer):void {
     super.showSubMenu(itemRenderer);
     if (currentOpenedSubMenu)
     {
      var itemSubmenu:AbstractSubMenu = currentOpenedSubMenu as AbstractSubMenu;
        itemSubmenu.setSize(Math.max(itemRenderer.getDimensions().width, itemSubmenu.buttonsWidth), 0);
        if (currentOpenedSubMenu is MainSubMenu)
      {
       var submenu:MainSubMenu = currentOpenedSubMenu as MainSubMenu;
       submenu.setTopPadding(itemRenderer.getDimensions().height);
        }
      //drop shadow for opened submenu
      currentOpenedSubMenu.filters = [new DropShadowFilter(1, 45, 0x000000, 1, 3, 3)];
     }
    }
   }
  }

The class of main menu button renderer MainMenuItemRenderer

  package modules
  {
   import com.moto.template.common.utils.MotoUtils;
   import com.moto.template.shell.model.vo.PropertyVO;
   import flash.display.MovieClip;
   import flash.events.Event;
   import flash.geom.Rectangle;
   import flash.text.TextField;
   import flash.text.TextFieldAutoSize;
     public class MainMenuItemRenderer extends AbstractMenuButtonItemRenderer
   {
    private var _tf:TextField;
    private var _button:MovieClip;
    private var _clip:MovieClip;
      public function MainMenuItemRenderer()
    {
     super();
       // Get slot elements
     _clip = getChildByName("clip") as MovieClip;
     _button = getChildByName("button") as MovieClip;
     _tf = simpleButtonText.textRead.getChildByName("tf") as TextField;
       _tf.autoSize = TextFieldAutoSize.LEFT;
       hitArea = _button;
    }
      override public function updateRenderer(data:Object):void
    {
     super.updateRenderer(data);
       var labelPropertyID:PropertyVO = data.getPropertyByID(1);
     _tf.embedFonts = MotoUtils.convertToBoolean(labelPropertyID.getParameter("embedFonts"));
     _tf.htmlText = String(labelPropertyID.value);
       var thisDimensions:Rectangle = getDimensions();
     var newWidth:Number = thisDimensions.width;
     var newHeight:Number = thisDimensions.height;
     var padding:Number = 10;
       if (_tf.width + padding > newWidth)
      newWidth = _tf.width + padding;
       if (_tf.height + padding > newHeight)
      newHeight = _tf.height + padding;
       setSize(newWidth, newHeight);
    }
      override public function getDimensions():Rectangle
    {
     return new Rectangle(0, 0, _button.width, _button.height);
    }
      override public function setSize(newWidth:Number, newHeight:Number):void
    {
     _button.width = newWidth;
     _button.height = newHeight;
       _clip.width = Math.round(newWidth);
     _clip.height = Math.round(newHeight);
       _tf.x = Math.round((newWidth - _tf.textWidth) 5) - 1;
     _tf.y = Math.round((newHeight - _tf.textHeight) .5) - 2;
    }
      override protected function over():void
    {
     _clip.gotoAndPlay(overLabel);
     simpleButtonText.gotoAndPlay(overLabel);
    }
      override protected function out():void
    {
     _clip.gotoAndPlay(outLabel);
     simpleButtonText.gotoAndPlay(outLabel);
    }
   }
    }

The class of submenu implementation MainSubMenu

  package modules
  {
   import caurina.transitions.Equations;
   import caurina.transitions.Tweener;
   import com.moto.template.common.events.ItemRendererEvent;
   import com.moto.template.common.Moto;
   import com.moto.template.common.utils.MotoUtils;
   import com.moto.template.modules.menu.AbstractMenuItemRenderer;
   import com.moto.template.modules.menu.DynamicMotoMenu;
   import flash.display.MovieClip;
   import flash.display.Sprite;
   import flash.events.Event;
   import flash.geom.Point;
   import flash.geom.Rectangle;
     public class MainSubMenu extends DynamicMotoMenu
   {
    static public const SUBMENU_ADDED:String = 'submenuAdded';
    private var _isHiding:Boolean = false;
    private var _SUBMENU_PADDING:int = 1;
    private var _CONTENT_PADDING:int = 12;
    private var _mask_mc:MovieClip;
    private var _back_mc:MovieClip;
    private var _back_body_mc:MovieClip;
    private var _itemsHolder:MovieClip;
    private var _submenuHolder:MovieClip;
    private var _contentHolder:MovieClip;
    private var _buttonsWidth:Number = 0;
    private var _buttonsHeight:Number = 0;
    private var _delimers:Array;
      public function MainSubMenu()
    {
     super();
       // Get elements
     _mask_mc = this['mask_mc'];
     _submenuHolder = this['submenuHolder'];
     _contentHolder = this['contentHolder'];
     _back_mc = _contentHolder['back_mc'];
     _back_body_mc = _back_mc['body_mc'];
     _itemsHolder = _contentHolder['itemsHolder'];
       _contentHolder.mask = _mask_mc;
     _itemsHolder.y = _CONTENT_PADDING;
       // Set sub menu remove delay
     subMenuRemoveDelay = 1;
       // Configure items arranger
     itemsArranger.autoUpdate = false;
     itemsArranger.columns = 1;
     itemsArranger.itemRendererCreationDelay = 0;
     itemsArranger.target = _itemsHolder;
      }
    override protected function initListControls(dataProvider:Array):void
   {
    super.initListControls(dataProvider);
       _initializationSuccess();
       if (!limitedMode)
     {
      updateSelectedMenuButton(Moto.getInstance().currentPage);
     }
   }
      protected function _initializationSuccess():void
    {
     var itemDelimer_mc:MovieClip;
     if (_delimers)
     {
      while (_delimers.length)
      {
       itemDelimer_mc = _delimers.shift();
       if (itemDelimer_mc.parent)
        itemDelimer_mc.parent.removeChild(itemDelimer_mc)
      }
     }
       _delimers = new Array();
       var item_btn:AbstractMenuItemRenderer;
     var length:int = itemsArranger.itemsCollection.length;
     var i:int = length;
       while (i--)
     {
      item_btn = itemsArranger.itemsCollection[i];
      _buttonsWidth = Math.max(_buttonsWidth, item_btn.getDimensions().width);
     }
       var nextY:Number = itemsArranger.yOffset;
     for (i = 0; i < length; i++)
     {
      item_btn = itemsArranger.itemsCollection[i];
      item_btn.y = nextY;
      nextY += item_btn.getDimensions().height + 0;
     }
       _buttonsHeight = nextY - 0;
       _back_body_mc.width = _buttonsWidth;
     _back_body_mc.height = _buttonsHeight + _itemsHolder.y + _CONTENT_PADDING;
     _mask_mc.width = _back_body_mc.width;
     _mask_mc.height = _back_body_mc.height;
       _contentHolder.y = -getDimensions().height;
    }
      public function setTopPadding(value:Number):void
    {
     _itemsHolder.y = value;
       _back_body_mc.height = buttonsHeight + _itemsHolder.y + _CONTENT_PADDING;
     _mask_mc.height = _back_body_mc.height;
    }
      override public function getItemRendererSubMenuPoint(itemRenderer:AbstractMenuItemRenderer):Point
    {
     return new Point(getDimensions().right + _SUBMENU_PADDING, itemRenderer.y + _itemsHolder.y);
    }
      override public function getSubMenuTargetHolder():Sprite
    {
     return _submenuHolder;
    }
      override public function getDimensions():Rectangle
    {
     return new Rectangle(0, 0, _back_body_mc.width, _back_body_mc.height);
    }
      override public function setSize(newWidth:Number, newHeight:Number):void
    {
     _back_body_mc.width = newWidth;
     _mask_mc.width = newWidth;
       var item_btn:AbstractMenuItemRenderer;
     var itemDimensions:Rectangle;
     var i:int = itemsArranger.itemsCollection.length;
     while (i--)
     {
      item_btn = itemsArranger.itemsCollection[i];
      itemDimensions = item_btn.getDimensions();
      item_btn.setSize(newWidth, itemDimensions.height);
     }
    }
      override public function show():void
    {
     super.show();
     Tweener.addTween(_contentHolder, { y: 0, time:.7, transition:Equations.easeOutCubic } );
    }
      override public function hide():void
    {
     if (!_isHiding)
     {
      _isHiding = true;
      super.hide();
      gotoAndPlay("hide");
      Tweener.addTween(_contentHolder, { y: -getDimensions().height, time:.7, transition:Equations.easeOutSine } );
     }
    }
      override protected function itemClickHandler(event:ItemRendererEvent):void
    {
     // Do not allow clicks while animation is playing
     if (!Moto.getInstance().animationIsPlaying)
      super.itemClickHandler(event);
    }
      override public function showSubMenu(itemRenderer:AbstractMenuItemRenderer):void
    {
     super.showSubMenu(itemRenderer);
     if (currentOpenedSubMenu)
     {
      var itemSubmenu:AbstractSubMenu = currentOpenedSubMenu as AbstractSubMenu;
        var point:Point = itemSubmenu.localToGlobal(new Point(itemSubmenu.getDimensions().right));
      if (point.x > stage.stageWidth)
       itemSubmenu.x = - getDimensions().width -  itemSubmenu.getDimensions().width - _SUBMENU_PADDING 2;
      dispatchEvent(new Event(SUBMENU_ADDED, true, true));
     }
    }
      public function get buttonsWidth():Number { return _buttonsWidth; }
      public function get buttonsHeight():Number { return _buttonsHeight; }
   }
    }
