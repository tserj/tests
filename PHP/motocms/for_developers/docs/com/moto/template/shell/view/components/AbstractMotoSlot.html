Example slot of the ReadMoreSlot button

  <slot id="+1 to the previous id" librarySymbolLinkage="ReadMoreSlot" animated="true" resizable="true" locked="false" mainPropertyID="1">

   <properties>

     <name><![CDATA[Text]]></name>
    </property>
   </properties>
   <template alpha="1" buttonMode="true" click="">
    <properties>
     <item propertyType="1"><![CDATA[<font face="standard 07_55" color="#282929" size="8">BUTTON LABEL</font>]]></item>
    </properties>
   </template>
  </slot>
  
  package slots
  {
   import com.moto.template.common.utils.MotoUtils;
   import com.moto.template.shell.model.vo.MotoObjectVO;
   import com.moto.template.shell.model.vo.PropertyVO;
   import com.moto.template.shell.view.components.AbstractMotoSlot;
     import flash.geom.Rectangle;
   import flash.events.Event;
   import flash.events.MouseEvent;
   import flash.display.MovieClip;
   import flash.text.TextField;
   import flash.text.TextFieldAutoSize;
     public class ReadMoreSlot extends AbstractMotoSlot
   {
    private var _tf:TextField;
    private var _button:MovieClip;
    private var _buttonMask:MovieClip;
    private var _bg:MovieClip;
      public function ReadMoreSlot()
    {
     super();
       // Get slot elements
     _buttonMask = getChildByName("buttonMask") as MovieClip;
     _bg = getChildByName("bg") as MovieClip;
     _button = getChildByName("button") as MovieClip;
     _tf = textHolder.getChildByName("tf") as TextField;
       _tf.autoSize = TextFieldAutoSize.LEFT;
     _tf.wordWrap = false;
     _tf.multiline = false;
       buttonMode = true;
     mouseChildren = false;
     hitArea = _button;
    }
      override public function init(vo:MotoObjectVO):void
    {
     super.init(vo);
       // Add the code here if you need to hande initialization
    }
      override public function updateProperty(property:PropertyVO):void
    {
     switch (property.propertyType)
     {
      case 1 :
       _tf.embedFonts = MotoUtils.convertToBoolean(property.parameters["embedFonts"]);
       _tf.htmlText = String(property.value);
         _tf.y = Math.ceil((_bg.clip.height - _tf.textHeight) /2 ) - 2;
         setSize(_tf.textWidth+10, getDimensions().height);
      break;
     }
    }
      override public function setSize(newWidth:Number, newHeight:Number):void
    {
     super.setSize(newWidth, newHeight);
       _buttonMask.clip.width = newWidth;
     _bg.clip.width = newWidth;
     _button.clip.width = newWidth;
    }
      override public function getDimensions():Rectangle
    {
     return new Rectangle(0, 0, _bg.clip.width, _bg.clip.height);
    }
      override protected function mouseOverHandler(event:MouseEvent):void {
     gotoAndPlay(overFrameLabel);
    }
      override protected function mouseOutHandler(event:MouseEvent):void {
     gotoAndPlay(outFrameLabel);
    }
   }
  }
